Zapytanie 1: Lista zwierząt z gatunkiem, klatką i strefą
SELECT zwierzeta.id_zwierzecia,
    CONCAT(zwierzeta.imie, ' czyli ', gatunki.nazwa_gatunku) AS zwierze,
    klatki.nazwa_klatki, 
    strefy.nazwa_strefy, 
    zwierzeta.waga_kg, 
    zwierzeta.status
FROM zwierzeta
	JOIN gatunki ON zwierzeta.id_gatunku = gatunki.id_gatunku
	JOIN klatki ON zwierzeta.id_klatki = klatki.id_klatki
	JOIN strefy ON klatki.id_strefy = strefy.id_strefy
ORDER BY strefy.nazwa_strefy, zwierzeta.imie;


Zapytanie 2: Łączne przychody z biletów wg rodzaju
SELECT rodzaje_biletow.id_rodzaju,
	rodzaje_biletow.nazwa_rodzaju,
    COUNT(*) AS liczba_transakcji,
    SUM(bilety.ilosc_biletow) AS sprzedane_bilety,
    SUM(bilety.cena_laczna) AS przychod_total
FROM bilety 
	JOIN rodzaje_biletow ON bilety.id_rodzaju = rodzaje_biletow.id_rodzaju
GROUP BY rodzaje_biletow.id_rodzaju, rodzaje_biletow.nazwa_rodzaju
ORDER BY przychod_total DESC;


Zapytanie 3: Ranking pracowników według liczby wykonanych karmień 
SELECT pracownicy.id_pracownika,
    CONCAT(pracownicy.imie, ' ', pracownicy.nazwisko) AS pracownik,
    pracownicy.stanowisko,
    COUNT(karmienie.id_karmienia) AS liczba_karmien,
    ROUND(SUM(karmienie.ilosc_kg), 2) AS kg_lacznie
FROM pracownicy
    JOIN karmienie ON pracownicy.id_pracownika = karmienie.id_pracownika
GROUP BY pracownicy.id_pracownika
ORDER BY liczba_karmien DESC;


Zapytanie 4: Zwierzęta wymagające wizyty weterynaryjnej
SELECT zwierzeta.id_zwierzecia,
    CONCAT(zwierzeta.imie, ' czyli ', gatunki.nazwa_gatunku) AS zwierze,
    MAX(wizyty_weterynaryjne.nastepna_wizyta) AS zaplanowana_wizyta
FROM zwierzeta
    JOIN gatunki ON zwierzeta.id_gatunku = gatunki.id_gatunku
    JOIN wizyty_weterynaryjne ON zwierzeta.id_zwierzecia = wizyty_weterynaryjne.id_zwierzecia
WHERE wizyty_weterynaryjne.nastepna_wizyta <= DATE_ADD(CURDATE(), INTERVAL 30 DAY)
GROUP BY zwierzeta.id_zwierzecia
ORDER BY zaplanowana_wizyta;

Zapytanie 5: Dzienny przychód ze sprzedaży biletów (GROUP BY DATE)
SELECT DATE(data_zakupu) AS dzien,
    COUNT(*) AS transakcje,
    SUM(ilosc_biletow) AS bilety,
    SUM(cena_laczna) AS przychod
FROM bilety
GROUP BY DATE(data_zakupu)
ORDER BY dzien;

Zapytanie 6: Średnia waga zwierząt wg typu diety (GROUP BY + AVG)
SELECT gatunki.typ_diety,
    COUNT(zwierzeta.id_zwierzecia) AS liczba_zwierzat,
    ROUND(AVG(zwierzeta.waga_kg), 2) AS srednia_waga_kg,
    ROUND(MAX(zwierzeta.waga_kg), 2) AS max_waga_kg
FROM zwierzeta
    JOIN gatunki ON zwierzeta.id_gatunku = gatunki.id_gatunku
WHERE zwierzeta.status = 'aktywne'
GROUP BY gatunki.typ_diety;


Zapytanie 7: Ranking zwierząt według wagi w obrębie gatunku
SELECT zwierzeta.imie,
    gatunki.nazwa_gatunku,
    zwierzeta.waga_kg,
    RANK() OVER (PARTITION BY zwierzeta.id_gatunku ORDER BY zwierzeta.waga_kg DESC) AS miejsce_wedlug_gatunku
FROM zwierzeta
    JOIN gatunki ON zwierzeta.id_gatunku = gatunki.id_gatunku;


Zapytanie 8: Najpopularniejsze atrakcje według liczby rezerwacji
SELECT atrakcje_dodatkowe.id_atrakcji,
    atrakcje_dodatkowe.nazwa_atrakcji,
    COUNT(rezerwacje_atrakcji.id_rezerwacji) AS rezerwacje,
    SUM(rezerwacje_atrakcji.ilosc_osob) AS uczestnicy_lacznie,
    atrakcje_dodatkowe.cena_za_osobe,
    ROUND(SUM(rezerwacje_atrakcji.ilosc_osob) * atrakcje_dodatkowe.cena_za_osobe, 2) AS przychod
FROM atrakcje_dodatkowe
    JOIN rezerwacje_atrakcji ON atrakcje_dodatkowe.id_atrakcji = rezerwacje_atrakcji.id_atrakcji
GROUP BY atrakcje_dodatkowe.id_atrakcji
ORDER BY rezerwacje DESC;


Zapytanie 9: Zwierzęta, które nie były leczone w ciągu ostatnich 6 miesięcy
SELECT zwierzeta.id_zwierzecia,
    zwierzeta.imie,
    gatunki.nazwa_gatunku,
    zwierzeta.data_przybycia
FROM zwierzeta
    JOIN gatunki ON zwierzeta.id_gatunku = gatunki.id_gatunku
WHERE zwierzeta.id_zwierzecia NOT IN (
    SELECT DISTINCT id_zwierzecia
    FROM wizyty_weterynaryjne
    WHERE data_wizyty >= DATE_SUB(CURDATE(), INTERVAL 6 MONTH)
);


Zapytanie 10: Łączna wartość sprzedanych biletów
SET @total := (SELECT SUM(cena_laczna) FROM bilety);
SELECT @total AS laczny_przychod_biletow,
       COUNT(*) AS liczba_transakcji,
       ROUND(@total / COUNT(*), 2) AS srednia_na_transakcje
FROM bilety;
